

\section{ARL Calculation}

\begin{itemize}
\item The Probability of obtaining the first out of control message on the $n-$th subgroup is $p(1-p)^{n-1}$ ( with n=1,2,\ldots$) which is the general form of the geometric distribution.

\item $f(n) = p(1-p)^{n-1}$  we need the expected value of $n$.

\item \[ E(n) =  \sum^\infty_{n=1} np(1-p)^{n-1} = p \sum^\infty_{n=1}n (1-p)^{n-1}\]

\item Using the fact that  $\sum^\infty_{n=1} (1-p)^{n-1}$  is an infinite geometric series such that
\[  \sum^\infty_{n=1} (1-p)^{n-1} = \frac{1}{p} \]

\end{itemize}
